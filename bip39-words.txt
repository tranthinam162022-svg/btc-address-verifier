def load_words_from_file(filename: str) -> List[str]:
    """Load words from a file."""
    with open(filename, 'r') as file:
        return [word.strip().lower() for word in file.readlines()]

def generate_bip39_mnemonic_from_words(words: List[str]) -> str:
    """Generate a BIP39 mnemonic from a list of words."""
    if len(words) != 12 and len(words) != 18 and len(words) != 24:
        raise ValueError("The number of words must be 12, 18, or 24.")
    return " ".join(words)

def generate_bip39_mnemonic_from_seed(seed: str) -> str:
    """Generate a BIP39 mnemonic from a seed."""
    # This is a simplified version. In practice, you would use a proper BIP39 implementation.
    # For example, you could use the `bip39` library.
    return " ".join(seed.split())

def main():
    """Main function."""
    if len(sys.argv) != 2:
        print("Usage: python gen_btc_eth_from_mnemonic.py --mnemonic \"nhập_chuỗi_mnemonic_của_bạn\"")
        sys.exit(1)

    mnemonic = sys.argv[1]

    # Load words from file
    words = load_words_from_file("words.txt")

    # Generate a BIP39 mnemonic from the words
    bip39_mnemonic = generate_bip39_mnemonic_from_words(words)

    # Generate a BIP39 mnemonic from the seed
    bip39_mnemonic_from_seed = generate_bip39_mnemonic_from_seed(mnemonic)

    print(f"BIP39 mnemonic from words: {bip39_mnemonic}")
    print(f"BIP39 mnemonic from seed: {bip39_mnemonic_from_seed}")

if __name__ == "__main__":
    main()
